class Member:
def **init**(self, member_id, name, email):
self.member_id = member_id
self.name = name
self.email = email

```
def update_email(self, new_email):
    self.email = new_email

def matches_id(self, member_id):
    return self.member_id == member_id

def to_csv_row(self):
    return [
        self.member_id,
        self.name,
        self.email
    ]

def __str__(self):
    return (
        f"Member ID: {self.member_id} | "
        f"Name: {self.name} | "
        f"Email: {self.email}"
    )
```
